\subsection{孤立奇点的类型}

\begin{frame}{一、问题引入——柯西积分公式的局限性}
	\onslide<+->
	回忆柯西积分公式：
	\onslide<+->
	\begin{theorem}[柯西积分公式]
		如果 $f(z)$ 在闭路 $C$ 及其内部区域 $D$ 内解析, 且 $z_0\in D$, 那么

		\vspace{5mm}
		$\qquad\qquad\displaystyle f^{(n)}(z)=\frac{n!}{2\pi i}\oint_C\alert{\frac{f(z)}{(z-z_0)^{n+1}}}\diff z.$
		\begin{center}
			\begin{tikzpicture}[overlay]
				\begin{tikzpicture}[framed,scale=.8,xshift=-35mm,yshift=-5mm]
				\filldraw[dcolorb,smooth,cstfill] plot coordinates {(-2.2,0) (-1.5,-0.7) (0,-1.1) (1,-1.1) (1.4,0) (0,1.2) (-1.5,1.2) (-2.2,0)};
				\draw[cstcurve,dcolora,smooth,cstarrowto] plot coordinates {(-.7,.8) (-1.1,.1) (-.7,-.4) (0.2,-.9) (0.5,-.4) (0.2,.3) (-.7,.8) };
				\fill[cstdot,dcolorc] (-.3,0) circle;
				\draw
					(0,.8) node[dcolora] {$C$}
					(1,-.6) node[dcolorb] {$D$}
					(0,0) node[dcolorc] {$z_0$};
				\end{tikzpicture}\end{tikzpicture}
		\end{center}
		\vspace{-3.5\baselineskip}
	\end{theorem}
	\onslide<+->
	直观地说, 柯西积分公式可以用来处理这种积分: \alert{被积函数在闭路 $C$ 内的奇点 $z_0$ 是由于被积函数分母中出现了因式 $z-z_0$ 导致的}.
\end{frame}

\begin{frame}{一、问题引入——柯西积分公式的局限性}
	\onslide<+->
	例如
	\begin{align*}
		\oint_{|z|=1} \frac{\cos z}{z}\diff z
		&\visible<+->{=2\pi i \cos z|_{z=0}}
		\visible<+->{=2\pi i,}\\
		\visible<+->{\oint_{|z-1|=1} \frac{z^3+1}{(z-1)^3}\diff z}
		&\visible<+->{=\frac{2\pi i}{2!}(z^3+1)''|_{z=1}}
		\visible<+->{=6\pi i.}
	\end{align*}
	\onslide<+->
	然而, 该公式无法处理诸如
	\[\frac1{2\pi i}\oint_{|z|=1} \frac{z}{\sin z}\diff z,\quad
	\frac1{2\pi i}\oint_{|z|=1} \frac{\cos z}{e^z-1}\diff z\]
	等类型的积分.
\end{frame}

\begin{frame}{一、问题引入——洛朗展开的应用}
	\onslide<+->
	回忆解析函数的洛朗展开形式: 
	\onslide<+->
	如果函数 $f(z)$ 在圆环域 $0<|z-z_0|<R$ 内解析, 那么 $f(z)$ 可以展开为洛朗级数
	\[f(z)=\sum_{n=-\infty}^{\infty} c_n(z-z_0)^n,\quad
	\visible<+->{\text{其中}\ c_n=\frac1{2\pi i}\oint_C\frac{f(z)}{(z-z_0)^{n+1}}\diff z,}\]
	\onslide<+->
	$C$ 是该圆环域内的闭路.

	\onslide<+->
	令 $n=-1$ 得到
	\[\aboxeq{\oint_C f(z)\diff z=2\pi i c_{-1}.}\]
	
	\onslide<2->
	\begin{tikzpicture}[overlay]
		\begin{tikzpicture}[scale=1.4,xshift=-50mm,yshift=-3mm,framed]
			\filldraw[dcolorb,cstfill] (-.3,0) circle (1.2);
			\draw[cstcurve,dcolora,smooth,cstarrowto,visible on=<3->] plot coordinates {(-.7,.8) (-1.1,.1) (-.7,-.4) (0.2,-.9) (0.5,-.4) (0.2,.3) (-.7,.8) };
			\filldraw[dcolorb,fill=white] (-.3,0) circle (.45);
			\fill[cstdot,dcolorc] (-.3,0) circle;
			\draw
			(0,.8) node[dcolora,visible on=<3->] {$C$}
			(-.3,-.3) node[dcolorc] {$z_0$};
		\end{tikzpicture}\end{tikzpicture}
		\vspace{-6.5\baselineskip}
\end{frame}

\begin{frame}{一、问题引入——洛朗展开的应用}
	\onslide<+->
	例如 $f(z)=\dfrac z{\sin z}$. 当 $0<|z|<\pi$ 时,
	\[f(z)=\frac z{z-\dfrac{z^3}{3!}+\dfrac{z^5}{5!}+\cdots}=1+\frac{z^2}6+\cdots\]
	\onslide<+->
	特别地, $c_{-1}=0$, 从而
	\[\oint_{|z|=1} f(z)\diff z=2\pi i c_{-1}=0.\]

	\onslide<+->
	这样便可求出这种类型的积分.
	\onslide<+->
	然而, 每次计算都需要求洛朗展开形式, 计算依然繁琐.
	\onslide<+->
	我们希望能通过一些\alert{较为简单的判定方法来得到 $c_{-1}$}, 而避免展开整个洛朗级数.
	\onslide<+->
	为了达到这个目的, 我们对奇点按在其洛朗展开的特点来进行分类.
\end{frame}


\begin{frame}{二、孤立奇点}
	\onslide<+->
	为了利用洛朗级数, 我们要求 $f(z)$ 在奇点 $z_0$ 的一个去心邻域 $0<|z-z_0|<\delta$ 内解析.
	\onslide<+->
	这样的奇点被称为\emph{孤立奇点}.

	\onslide<+->
	\begin{example}
			考虑函数 $f(z)=\dfrac1{\sin(1/z)}$, 显然 $0$ 和 $z_k=\dfrac1{k\pi}$ 是奇点, $k$ 是非零整数.
		\onslide<+->{因为 $\lim\limits_{k\to+\infty} z_k=0$, 所以 $0$ 的任何一个去心邻域内都有奇点.
		}\onslide<+->{此时 $0$ 不是孤立奇点.}
		\onslide<3->{
		\begin{center}
			\begin{tikzpicture}[framed,scale=.9]
				\draw[cstcurve,dcolorc] (0,0) circle (1.3);
				\fill[cstdot] (0,0) circle;
				\fill[cstdot,dcolorb] (2,0) circle;
				\fill[cstdot,dcolora] (1,0) circle;
				\fill[cstdot,dcolorb] (0.6667,0) circle;
				\fill[cstdot,dcolora] (0.5,0) circle;
				\fill[cstdot,dcolorb] (0.4,0) circle;
				\draw
					(-0.3,0) node {$0$}
					(2,-0.3) node[dcolorb] {$z_1$}
					(1,0.3) node[dcolora] {$z_2$}
					(0.6667,-0.3) node[dcolorb] {$z_3$}
					(0.5,0.3) node[dcolora] {$z_4$}
					(0.4,-0.3) node[dcolorb] {$z_k$};
			\end{tikzpicture}
		\end{center}}
	\end{example}
\end{frame}


\begin{frame}{二、孤立奇点}
	\onslide<+->
	\begin{example}
		\begin{itemize}
			\item $z=0$ 是 $e^{\frac1z},\dfrac{\sin z}z,\dfrac z{\sin z}$ 的孤立奇点.
			\item $z=-1$ 是 $\dfrac1{z(z+1)}$ 的孤立奇点.
			\item $z=0$ 不是 $\dfrac1{\sin(1/z)}$ 的孤立奇点.
		\end{itemize}
	\end{example}

	\onslide<+->
	若 $f(z)$ 只有有限多个奇点, 则这些奇点都是孤立奇点.
\end{frame}


\begin{frame}{三、可去奇点——定义}
	\onslide<+->
	\begin{definition}
		若 $f(z)$ 在孤立奇点 $z_0$ 的去心邻域的洛朗级数没有主要部分, 即
		\[f(z)=c_0+c_1(z-z_0)+c_2(z-z_0)^2+\cdots,\quad 0<|z-z_0|<\delta,\]
		是幂级数, 则称 $z_0$ 是 $f(z)$ 的\emph{可去奇点}.
	\end{definition}

	\onslide<+->
	设 $g(z)$ 为右侧幂级数的和函数, 则 $g(z)$ 在 $|z-z_0|<\delta$ 上解析,
	\onslide<+->
	且除 $z_0$ 外 $f(z)=g(z)$.
	\onslide<+->
	通过补充或修改定义 $f(z_0)=g(z_0)=c_0$, 可使得 $f(z)$ 也在 $z_0$ 解析.
	\onslide<+->
	这就是``\emph{可去}''的含义.

\end{frame}


\begin{frame}{三、可去奇点——判定方法}
	\onslide<+->
	如果 $z_0$ 是 $f(z)$ 的可去奇点, 那么
	\begin{itemize}
		\item $\lim\limits_{z\to z_0}f(z)=c_0$ 存在;
		\item $\lim\limits_{z\to z_0}(z-z_0)f(z)=0$.
	\end{itemize}
	\onslide<+->
	实际上, 这三者是等价的:
	\onslide<+->
	\begin{theorem}[可去奇点的判定方法]
		\begin{tabular}{rl}
			$z_0$ 是 $f(z)$ 的可去奇点
			&$\iff\lim\limits_{z\to z_0}f(z)$ 存在\\
			&$\iff\lim\limits_{z\to z_0}(z-z_0)f(z)=0$.
		\end{tabular}
	\end{theorem}
\end{frame}


\begin{frame}{例题: 可去奇点的判定}
	\onslide<+->
	\begin{example*}
		设 $f(z)=\dfrac{e^z-1}z$.
		\onslide<+->{那么\[\lim\limits_{z\to0}zf(z)=e^0-1=0.\]
		}\onslide<+->{因此 $0$ 是 $f(z)$ 的可去奇点.}

		\onslide<+->{实际上当 $0<|z|<+\infty$ 时,
		\[f(z)=\frac1z\left(z+\frac{z^2}{2!}+\frac{z^3}{3!}+\cdots\right)=1+\frac{z}{2!}+\frac{z^2}{3!}+\cdots\]}
	\end{example*}
\end{frame}


\begin{frame}{例题: 可去奇点的应用}
	\onslide<+->
	\begin{example*}
		设 $f(z)=\dfrac z{\sin z}$.
		\onslide<+->{则
		\[\lim_{z\to0}zf(z)=\lim_{z\to 0}\frac{z^2}{\sin z}
		\visible<+->{\djwqx\lim_{z\to 0}\frac{z^2}{z}=0.}\]
		}\onslide<+->{因此 $0$ 是 $f(z)$ 的可去奇点.}

		\onslide<+->{所以 $f(z)$ 在 $0<|z|<\pi$ 内的洛朗展开中 $c_{-1}=0$,}\onslide<+->{
			\[\oint_{|z|=1}f(z)\diff z=2\pi i c_{-1}=0.\qquad\qquad\qquad\]
		}
