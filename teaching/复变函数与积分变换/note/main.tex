\newcommand{\randpts}[3][10]{
  \foreach\i in {0,1,...,#1}{
    \pgfmathparse{rdv}\let\rdv\pgfmathresult
    \coordinate (\i) at ({#2*\rdv*cos(360/#1*\i)},{#3*\rdv*sin(360/#1*\i)});
  }
}
